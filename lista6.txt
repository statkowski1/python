#zad3
import numpy as np
def wektor(n):
    x = [2**n for n in range(1, n*n+1)]
    y = np.array(x)
    z = y.reshape((n, n))
    return z
print(wektor(5))

import numpy as np
def TablicaPoteg(podstawa, ilosc):
    a = np.logspace(1, ilosc, ilosc, base=podstawa, dtype='int')
    return a
print(TablicaPoteg(2, 5))

#zad5
import numpy as np
def MacierzDiagonalna(dlugosc):
    x = np.diag([a for a in range(dlugosc, 0, -1)], 2)
    return x
print(MacierzDiagonalna(4))
